\section{Phrase structure}
\begin{itemize}
\item In linguistics: the X$'$ schema
\Tree [.XP [.(Specifier) ] [.X\1 [.X Head ] [.(Complement) ] ] ]
\item In computation linguistics: the X$'$ schema is simplified
\Tree [.XP [.(Specifier) ] [.X Head ] [.(Complement) ] ]
\item Only lexical categories can project a phrase (XP)\\\newline
\Tree [.\w{NP} [.\w{NNP} Roger ] ]
\Tree [.\w{NP} [.\w{PRP} she ] ]
\Tree [.\w{WHNP} [.\w{WP} who ] ]
\Tree [.\w{VP} [.\w{VBD} ate ] ]
\Tree [.\w{ADJP} [.\w{JJ} tasty ] ]
\Tree [.\w{ADVP} [.\w{RB} slowly ] ]
\Tree [.\w{WHADVP} [.\w{WRB} when ] ]
\Tree [.\w{PP} [.\w{IN} about ] [.\w{NP} [.\w{NNP} Roger ] ] ]
\Tree [.\w{WHPP} [.\w{IN} on ] [.\w{WHNP} [.\w{WP} what ] ] ]
	\begin{itemize}
	\item Complement (has to be another XP): obligatory in some cases, and optional in others (but this can be ambiguous)\\
	\parbox[t]{.2\textwidth}{\ex
	\Tree [.VP [.VBP throw ] [.\w{NP} [.DT the ] [.NN ball ] ] !{\qframesubtree} ]
	\xe}%
	\parbox[t]{.35\textwidth}{\ex
	\Tree [.VP [.VBP give ] [.\w{NP} [.DT the ] [.NN ball ] ] !{\qframesubtree} [.\w{PP} [.TO to ] [.NP [.NN someone ] ] ] !{\qframesubtree} ]
	\xe}%
	\parbox[t]{.2\textwidth}{\ex
	\Tree [.ADJP [.JJ full ] [.\w{PP} [.IN of ] [.NP [.NN life ] ] ] !{\qframesubtree} ]
	\xe}%
	\parbox[t]{.2\textwidth}{\ex
	\Tree [.PP [.IN about ] [.NP [.NNP Roger ] ] !{\qframesubtree} ]
	\xe}
	\pagebreak
	\item Modification (always optional):
		\begin{enumerate}
		\item Premodifiers: generally are placed inside the phrase they are associated with, with the exception of some VP premodifiers (attached at S-level)\\
		\parbox[t]{.25\textwidth}{\ex\Tree [.NP [.DT the ] [.\w{JJ} red ] !{\qframesubtree} [.NN ball ] ]\xe}%
		\parbox[t]{.35\textwidth}{\ex\Tree [.NP [.\w{NP} [.DT a ] [.NN person ] [.POS 's ] ] !{\qframesubtree} [.NN money ] ]\xe}%
		\parbox[t]{.25\textwidth}{\ex\Tree [.ADJP [.\w{ADVP} [.RB extremely ] ] !{\qframesubtree} [.JJ delicious ] ]\xe}\\
		\parbox[t]{.25\textwidth}{\ex\Tree [.ADVP [.\w{NP} [.CD one ] [.NN year ] ] !{\qframesubtree} [.RB ago ] ]\xe}%
		\parbox[t]{.4\textwidth}{\ex\Tree [.S [.NP [.NNP Sandy ] ] [.\w{ADVP} [.RB often ] ] !{\qframesubtree} [.VP [.VBZ throws ] [.NP [.NNS curves ] ] ] ]\xe}
	\item Postmodifiers\\
	\parbox[t]{.4\textwidth}{\ex\Tree [.NP [.NP [.DT a ] [.NN teacher ] ] [.\w{PP} [.IN of ] [.NP [.NN chemistry ] ] ] !{\qframesubtree} ]\xe}%
	\parbox[t]{.25\textwidth}{\ex\Tree [.ADJP [.ADJP [.JJR taller ] ] [.\w{PP} [.IN than ] [.NP [.PRP him ] ] ] !{\qframesubtree} ]\xe}\\

	\end{enumerate}
	\end{itemize}
\end{itemize}

\subsection{SQ}
\begin{itemize}
\item Yes/No questions
\begin{itemize}
	\item \textit{Roger is$_i$ smart.} $\rightarrow$ \textit{Is$_i$ Roger $t_i$ smart?} (this is called the \textit{head-movement}, but we don't mark it in the Treebank parse)\\
	\parbox[t]{.4\textwidth}{\ex\Tree [.S \qroof{Roger}.NP [.VP [.VBZ is ] !{\qframesubtree} \qroof{smart}.ADJP ] ]\xe}%
	\parbox[t]{.4\textwidth}{\ex\Tree [.\w{SQ} [.VBZ Is ] !{\qframesubtree} \qroof{Roger}.NP \qroof{smart}.ADJP [.{.} ? ] ]\xe}
	\item \textit{Roger will$_i$ throw the ball.} $\rightarrow$ \textit{Will$_i$ Roger t$_i$ throw the ball?}\\
	\parbox[t]{.4\textwidth}{\ex\Tree [.S \qroof{Roger}.NP [.VP [.MD will ] !{\qframesubtree} \qroof{throw the ball}.VP ] ]\xe}%
	\parbox[t]{.4\textwidth}{\ex\Tree [.\w{SQ} [.MD Will ] !{\qframesubtree} \qroof{Roger}.NP \qroof{throw the ball}.VP [.{.} ? ] ]\xe}

	\parbox[t]{.4\textwidth}{\ex\Tree [.\w{SQ} [.VBD Did ] !{\qframesubtree} \qroof{Roger}.NP \qroof{throw the ball}.VP [.{.} ? ] ]\xe}

\subsection{SBARQ}
\begin{itemize}
\item Wh-questions (contain a gap and require a trace)
	\begin{itemize}
